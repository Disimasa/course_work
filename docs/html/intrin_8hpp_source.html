59 inline unsigned int trailingZeros32(unsigned int value) {
60 #if defined(_MSC_VER)
61 #if (_MSC_VER < 1700) || defined(_M_ARM) || defined(_M_ARM64)
62  unsigned long index = 0;
63  _BitScanForward(&index, value);
64  return (unsigned int)index;
65 #elif defined(__clang__)
66  // clang-cl doesn't export _tzcnt_u32 for non BMI systems
67  return value ? __builtin_ctz(value) : 32;
68 #else
69  return _tzcnt_u32(value);
70 #endif
71 #elif defined(__GNUC__) || defined(__GNUG__)
72  return __builtin_ctz(value);
73 #elif defined(__ICC) || defined(__INTEL_COMPILER)
74  return _bit_scan_forward(value);
75 #elif defined(__clang__)
76  return llvm.cttz.i32(value, true);
77 #else
78  static const int MultiplyDeBruijnBitPosition[32] = {
79  0, 1, 28, 2, 29, 14, 24, 3, 30, 22, 20, 15, 25, 17, 4, 8,
80  31, 27, 13, 23, 21, 19, 16, 7, 26, 12, 18, 6, 11, 5, 10, 9 };
81  return MultiplyDeBruijnBitPosition[((uint32_t)((value & -value) * 0x077CB531U)) >> 27];
82 #endif
83 }
84 }

239 // AVX2 can be used together with SSE2, so
240 // we define those two sets of intrinsics at once.
241 // Most of the intrinsics do not conflict (the proper overloaded variant is
242 // resolved by the argument types, e.g. v_float32x4 ~ SSE2, v_float32x8 ~ AVX2),
243 // but some of AVX2 intrinsics get v256_ prefix instead of v_, e.g. v256_load() vs v_load().
244 // Correspondingly, the wide intrinsics (which are mapped to the "widest"
245 // available instruction set) will get vx_ prefix
246 // (and will be mapped to v256_ counterparts) (e.g. vx_load() => v256_load())
247 #if CV_AVX2
248 
249 #define CV__SIMD_FORWARD 256
250 #include "opencv2/core/hal/intrin_forward.hpp"
251 #include "opencv2/core/hal/intrin_avx.hpp"
252 
253 #endif

256 // we define those sets of intrinsics at once.
257 // For some of AVX512 intrinsics get v512_ prefix instead of v_, e.g. v512_load() vs v_load().
258 // Wide intrinsics will be mapped to v512_ counterparts in this case(e.g. vx_load() => v512_load())
