86 class TLSDataAccumulator : public TLSData<T>
87 {
88  mutable cv::Mutex mutex;
89  mutable std::vector<T*> dataFromTerminatedThreads;
90  std::vector<T*> detachedData;
91  bool cleanupMode;
92 public:

94  ~TLSDataAccumulator()
95  {

106  void gather(std::vector<T*> &data) const
107  {
108  CV_Assert(cleanupMode == false); // state is not valid
109  CV_Assert(data.empty());
110  {
111  std::vector<void*> &dataVoid = reinterpret_cast<std::vector<void*>&>(data);
112  TLSDataContainer::gatherData(dataVoid);
113  }
114  {
115  AutoLock lock(mutex);
116  data.reserve(data.size() + dataFromTerminatedThreads.size());
117  for (typename std::vector<T*>::const_iterator i = dataFromTerminatedThreads.begin(); i != dataFromTerminatedThreads.end(); ++i)
118  {
119  data.push_back((T*)*i);
120  }
121  }
122  }
123 
130  std::vector<T*>& detachData()
131  {
132  CV_Assert(cleanupMode == false); // state is not valid
133  std::vector<void*> dataVoid;
134  {
135  TLSDataContainer::detachData(dataVoid);
136  }
137  {
138  AutoLock lock(mutex);
139  detachedData.reserve(dataVoid.size() + dataFromTerminatedThreads.size());
140  for (typename std::vector<T*>::const_iterator i = dataFromTerminatedThreads.begin(); i != dataFromTerminatedThreads.end(); ++i)
141  {
142  detachedData.push_back((T*)*i);
143  }
144  dataFromTerminatedThreads.clear();
145  for (typename std::vector<void*>::const_iterator i = dataVoid.begin(); i != dataVoid.end(); ++i)
146  {
147  detachedData.push_back((T*)(void*)*i);
148  }
149  }
150  dataVoid.clear();
151  return detachedData;
152  }
153 
155  void cleanupDetachedData()
156  {
157  AutoLock lock(mutex);
158  cleanupMode = true;
159  _cleanupDetachedData();
160  cleanupMode = false;
161  }
162 
164  void cleanup()
165  {
166  cleanupMode = true;
167  TLSDataContainer::cleanup();
168 
169  AutoLock lock(mutex);
170  _cleanupDetachedData();
171  _cleanupTerminatedData();
172  cleanupMode = false;
173  }
174 
176  void release()
177  {
178  cleanupMode = true;
179  TLSDataContainer::release();
180  {
181  AutoLock lock(mutex);
182  _cleanupDetachedData();
183  _cleanupTerminatedData();
184  }
185  }
186 
187 protected:
188  // synchronized
189  void _cleanupDetachedData()
190  {
191  for (typename std::vector<T*>::iterator i = detachedData.begin(); i != detachedData.end(); ++i)
192  {
193  deleteDataInstance((T*)*i);
194  }
195  detachedData.clear();
196  }
197 
198  // synchronized
199  void _cleanupTerminatedData()
200  {
201  for (typename std::vector<T*>::iterator i = dataFromTerminatedThreads.begin(); i != dataFromTerminatedThreads.end(); ++i)
202  {
203  deleteDataInstance((T*)*i);
204  }
205  dataFromTerminatedThreads.clear();
206  }
207 
208 protected:
209  virtual void* createDataInstance() const CV_OVERRIDE
210  {
211  // Note: we can collect all allocated data here, but this would require raced mutex locks
212  return new T;
213  }
214  virtual void deleteDataInstance(void* pData) const CV_OVERRIDE
215  {
216  if (cleanupMode)
217  {
218  delete (T*)pData;
219  }
220  else
221  {
222  AutoLock lock(mutex);
223  dataFromTerminatedThreads.push_back((T*)pData);
224  }
225  }
226 };
