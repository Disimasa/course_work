74 inline Rect& operator&=(Rect& lhs, const Rect& rhs)
75 {
76  int x1 = std::max(lhs.x, rhs.x);
77  int y1 = std::max(lhs.y, rhs.y);
78  lhs.width = std::min(lhs.x + lhs.width, rhs.x + rhs.width) - x1;
79  lhs.height = std::min(lhs.y + lhs.height, rhs.y + rhs.height) - y1;
80  lhs.x = x1;
81  lhs.y = y1;
82  if( lhs.width <= 0 || lhs.height <= 0 )
83  lhs = Rect();
84  return lhs;
85 }
