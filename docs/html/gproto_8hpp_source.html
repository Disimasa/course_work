81 template<typename Tg>
82 cv::GIOProtoArgs<Tg>& operator += (cv::GIOProtoArgs<Tg> &lhs, const cv::GIOProtoArgs<Tg> &rhs)
83 {
84  lhs.m_args.reserve(lhs.m_args.size() + rhs.m_args.size());
85  lhs.m_args.insert(lhs.m_args.end(), rhs.m_args.begin(), rhs.m_args.end());
