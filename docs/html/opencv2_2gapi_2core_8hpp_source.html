508  G_TYPED_KERNEL(
509  GKMeansND,
510  <std::tuple<GOpaque<double>,GMat,GMat>(GMat,int,GMat,TermCriteria,int,KmeansFlags)>,
511  "org.opencv.core.kmeansND") {
512 
513  static std::tuple<GOpaqueDesc,GMatDesc,GMatDesc>
514  outMeta(const GMatDesc& in, int K, const GMatDesc& bestLabels, const TermCriteria&, int,
515  KmeansFlags flags) {
516  GAPI_Assert(in.depth == CV_32F);
517  std::vector<int> amount_n_dim = detail::checkVector(in);
518  int amount = amount_n_dim[0], dim = amount_n_dim[1];
519  if (amount == -1) // Mat with height != 1, width != 1, channels != 1 given
520  { // which means that kmeans will consider the following:
521  amount = in.size.height;
522  dim = in.size.width * in.chan;
523  }
524  // kmeans sets these labels' sizes when no bestLabels given:
525  GMatDesc out_labels(CV_32S, 1, Size{1, amount});
526  // kmeans always sets these centers' sizes:
527  GMatDesc centers (CV_32F, 1, Size{dim, K});
528  if (flags & KMEANS_USE_INITIAL_LABELS)
529  {
