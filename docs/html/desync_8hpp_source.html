29  // An universal yield for desync.
30  // Yields output objects according to the input Types...
31  // Reuses gkernel machinery.
32  // FIXME: This function can be generic and declared in gkernel.hpp
33  // (it is there already, but a part of GKernelType[M]
34  template<typename... R, int... IIs>
35  static std::tuple<R...> yield(cv::GCall &call, cv::detail::Seq<IIs...>) {
36  return std::make_tuple(cv::detail::Yield<R>::yield(call, IIs)...);
37  }

40 template<typename G>
41 G desync(const G &g) {
42  cv::GKernel k{
43  GDesync::id() // kernel id
44  , "" // kernel tag
45  , [](const GMetaArgs &a, const GArgs &) {return a;} // outMeta callback
46  , {cv::detail::GTypeTraits<G>::shape} // output Shape
47  , {cv::detail::GTypeTraits<G>::op_kind} // input data kinds
48  , {cv::detail::GObtainCtor<G>::get()} // output template ctors
49  };
50  cv::GCall call(std::move(k));
51  call.pass(g);
52  return std::get<0>(GDesync::yield<G>(call, cv::detail::MkSeq<1>::type()));
53 }
