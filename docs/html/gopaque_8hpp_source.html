111  // This class represents a typed object reference.
112  // Depending on origins, this reference may be either "just a" reference to
113  // an object created externally, OR actually own the underlying object
114  // (be value holder).
115  class BasicOpaqueRef
116  {
117  public:
118  cv::GOpaqueDesc m_desc;
119  virtual ~BasicOpaqueRef() {}

122  virtual const void* ptr() const = 0;
123  virtual void set(const cv::util::any &a) = 0;

126  template<typename T> class OpaqueRefT final: public BasicOpaqueRef
127  {
128  using empty_t = util::monostate;
129  using ro_ext_t = const T *;
130  using rw_ext_t = T *;
131  using rw_own_t = T ;
132  util::variant<empty_t, ro_ext_t, rw_ext_t, rw_own_t> m_ref;
133 
134  inline bool isEmpty() const { return util::holds_alternative<empty_t>(m_ref); }
135  inline bool isROExt() const { return util::holds_alternative<ro_ext_t>(m_ref); }
136  inline bool isRWExt() const { return util::holds_alternative<rw_ext_t>(m_ref); }
137  inline bool isRWOwn() const { return util::holds_alternative<rw_own_t>(m_ref); }
138 
139  void init(const T* obj = nullptr)
140  {
141  if (obj) m_desc = cv::descr_of(*obj);
142  }
143 
144  public:
145  OpaqueRefT() { init(); }
146  virtual ~OpaqueRefT() {}
147 
148  explicit OpaqueRefT(const T& obj) : m_ref(&obj) { init(&obj); }
149  explicit OpaqueRefT( T& obj) : m_ref(&obj) { init(&obj); }
150  explicit OpaqueRefT( T&& obj) : m_ref(std::move(obj)) { init(&obj); }
151 
152  // Reset a OpaqueRefT. Called only for objects instantiated
153  // internally in G-API (e.g. temporary GOpaque<T>'s within a
154  // computation). Reset here means both initialization
155  // (creating an object) and reset (discarding its existing
156  // content before the next execution). Must never be called
157  // for external OpaqueRefTs.
158  void reset()
159  {
160  if (isEmpty())
161  {
162  T empty_obj{};
163  m_desc = cv::descr_of(empty_obj);
164  m_ref = std::move(empty_obj);
165  GAPI_Assert(isRWOwn());
166  }
167  else if (isRWOwn())
168  {
169  util::get<rw_own_t>(m_ref) = {};
170  }
171  else GAPI_Assert(false); // shouldn't be called in *EXT modes
172  }
173 
174  // Obtain a WRITE reference to underlying object
175  // Used by CPU kernel API wrappers when a kernel execution frame
176  // is created
177  T& wref()
178  {
179  GAPI_Assert(isRWExt() || isRWOwn());
180  if (isRWExt()) return *util::get<rw_ext_t>(m_ref);
181  if (isRWOwn()) return util::get<rw_own_t>(m_ref);
182  util::throw_error(std::logic_error("Impossible happened"));
183  }
184 
185  // Obtain a READ reference to underlying object
186  // Used by CPU kernel API wrappers when a kernel execution frame
187  // is created
188  const T& rref() const
189  {
190  // ANY object can be accessed for reading, even if it declared for
191  // output. Example -- a GComputation from [in] to [out1,out2]
192  // where [out2] is a result of operation applied to [out1]:
193  //
194  // GComputation boundary
195  // . . . . . . .
196  // . .
197  // [in] ----> foo() ----> [out1]
198  // . . :
199  // . . . .:. . .
200  // . V .
201  // . bar() ---> [out2]
202  // . . . . . . . . . . . .
203  //
204  if (isROExt()) return *util::get<ro_ext_t>(m_ref);
205  if (isRWExt()) return *util::get<rw_ext_t>(m_ref);
206  if (isRWOwn()) return util::get<rw_own_t>(m_ref);
207  util::throw_error(std::logic_error("Impossible happened"));
208  }

210  virtual void mov(BasicOpaqueRef &v) override {
211  OpaqueRefT<T> *tv = dynamic_cast<OpaqueRefT<T>*>(&v);
212  GAPI_Assert(tv != nullptr);
213  wref() = std::move(tv->wref());
214  }
215 
216  virtual const void* ptr() const override { return &rref(); }
217 
218  virtual void set(const cv::util::any &a) override {
219  wref() = util::any_cast<T>(a);
220  }
221  };

223  // This class strips type information from OpaqueRefT<> and makes it usable
224  // in the G-API executables (carrying run-time data/information to kernels).
225  // Part of GRunArg.
226  // Its methods are typed proxies to OpaqueRefT<T>.
227  // OpaqueRef maintains "reference" semantics so two copies of OpaqueRef refer
228  // to the same underlying object.
229  class OpaqueRef
230  {
231  std::shared_ptr<BasicOpaqueRef> m_ref;
232  cv::detail::OpaqueKind m_kind;
233 
234  template<typename T> inline void check() const
235  {
236  GAPI_DbgAssert(dynamic_cast<OpaqueRefT<T>*>(m_ref.get()) != nullptr);
237  }
238 
239  public:
240  OpaqueRef() = default;
241 
242  template<
243  typename T,
244  typename = util::are_different_t<OpaqueRef, T>
245  >
246  // FIXME: probably won't work with const object
247  explicit OpaqueRef(T&& obj) :
248  m_ref(new OpaqueRefT<util::decay_t<T>>(std::forward<T>(obj))),
249  m_kind(GOpaqueTraits<util::decay_t<T>>::kind) {}
250 
251  cv::detail::OpaqueKind getKind() const
252  {
253  return m_kind;
254  }
255 
256  template<typename T> void reset()
257  {
258  if (!m_ref) m_ref.reset(new OpaqueRefT<T>());
259  check<T>();
260  storeKind<T>();
261  static_cast<OpaqueRefT<T>&>(*m_ref).reset();
262  }
263 
264  template <typename T>
265  void storeKind()
266  {
267  m_kind = cv::detail::GOpaqueTraits<T>::kind;
268  }
269 
270  template<typename T> T& wref()
271  {
272  check<T>();
273  return static_cast<OpaqueRefT<T>&>(*m_ref).wref();
274  }
275 
276  template<typename T> const T& rref() const
277  {
278  check<T>();
279  return static_cast<OpaqueRefT<T>&>(*m_ref).rref();
280  }
281 
282  void mov(OpaqueRef &v)
283  {
284  m_ref->mov(*v.m_ref);
285  }
286 
287  cv::GOpaqueDesc descr_of() const
288  {
289  return m_ref->m_desc;
290  }
291 
292  // May be used to uniquely identify this object internally
293  const void *ptr() const { return m_ref->ptr(); }
294 
295  // Introduced for in-graph meta handling
296  OpaqueRef& operator= (const cv::util::any &a)
297  {
298  m_ref->set(a);
299  return *this;
300  }
301  };

308 template<typename T> class GOpaque
309 {
310 public:
311  // Host type (or Flat type) - the type this GOpaque is actually
312  // specified to.
313  using HT = typename detail::flatten_g<util::decay_t<T>>::type;
314 
315  GOpaque() { putDetails(); } // Empty constructor
316  explicit GOpaque(detail::GOpaqueU &&ref) // GOpaqueU-based constructor
317  : m_ref(ref) { putDetails(); } // (used by GCall, not for users)
318 
320  detail::GOpaqueU strip() const {
321  return m_ref;
322  }
324  static void Ctor(detail::OpaqueRef& ref) {
325  ref.reset<HT>();
326  }
327 private:
328  void putDetails() {
329  m_ref.setConstructFcn(&Ctor);
330  m_ref.specifyType<HT>();
331  m_ref.storeKind<HT>();
332  }
333 
334  detail::GOpaqueU m_ref;
335 };
