489 {
490  double operator ()(const Matx<_Tp, m, m>& a) const
491  {
492  Matx<_Tp, m, m> temp = a;
493  double p = LU(temp.val, m*sizeof(_Tp), m, 0, 0, 0);
494  if( p == 0 )
495  return p;
496  for( int i = 0; i < m; i++ )
497  p *= temp(i, i);
498  return p;
499  }
