161 template<typename K, typename V>
162 IOStream& operator<< (IOStream& os, const std::map<K, V> &m) {
163  const uint32_t sz = static_cast<uint32_t>(m.size());
164  os << sz;
165  for (const auto& it : m) os << it.first << it.second;
166  return os;
167 }
168 template<typename K, typename V>
169 IIStream& operator>> (IIStream& is, std::map<K, V> &m) {
170  m.clear();
171  uint32_t sz = 0u;
172  is >> sz;
173  for (std::size_t i = 0; i < sz; ++i) {
174  K k{};
175  V v{};
176  is >> k >> v;
177  m[k] = v;
178  }
179  return is;
180 }

184  os << sz;

202 IOStream& operator<< (IOStream& os, const std::vector<T> &ts) {
203  const uint32_t sz = static_cast<uint32_t>(ts.size());
204  os << sz;

208 template<typename T>
209 IIStream& operator>> (IIStream& is, std::vector<T> &ts) {
210  uint32_t sz = 0u;
211  is >> sz;
212  if (sz == 0u) {
213  ts.clear();
214  }
215  else {
216  ts.resize(sz);
217  for (std::size_t i = 0; i < sz; ++i) is >> ts[i];
218  }
219  return is;
220 }

228 template<typename V, typename X, typename... Xs>
229 IOStream& put_v(IOStream& os, const V& v, std::size_t x) {
230  return (x == 0u)
231  ? os << cv::util::get<X>(v)
232  : put_v<V, Xs...>(os, v, x-1);
233 }
234 template<typename V>
235 IIStream& get_v(IIStream&, V&, std::size_t, std::size_t) {
236  GAPI_Assert(false && "variant<<: requested index is invalid");
237 }
238 template<typename V, typename X, typename... Xs>
239 IIStream& get_v(IIStream& is, V& v, std::size_t i, std::size_t gi) {
240  if (i == gi) {
241  X x{};
242  is >> x;
243  v = V{std::move(x)};
244  return is;
245  } else return get_v<V, Xs...>(is, v, i+1, gi);
246 }

280 struct try_deserialize_comparg<std::tuple<T, Types...>> {
281 static cv::util::optional<GCompileArg> exec(const std::string& tag, cv::gapi::s11n::IIStream& is) {
282  if (tag == cv::detail::CompileArgTag<T>::tag()) {
283  static_assert(cv::gapi::s11n::detail::has_S11N_spec<T>::value,
284  "cv::gapi::deserialize<GCompileArgs, Types...> expects Types to have S11N "
285  "specializations with deserialization callbacks!");
286  return cv::util::optional<GCompileArg>(
287  GCompileArg { cv::gapi::s11n::detail::S11N<T>::deserialize(is) });
288  }
289  return try_deserialize_comparg<std::tuple<Types...>>::exec(tag, is);
290 }

295 template<typename RMatAdapterType>
296 struct deserialize_runarg {
297 static GRunArg exec(cv::gapi::s11n::IIStream& is, uint32_t idx) {
298  if (idx == GRunArg::index_of<RMat>()) {
299  auto ptr = std::make_shared<RMatAdapterType>();
300  ptr->deserialize(is);
301  return GRunArg { RMat(std::move(ptr)) };
302  } else { // non-RMat arg - use default deserialization
303  GRunArg arg;
304  getRunArgByIdx(is, arg, idx);
305  return arg;
306  }
307 }

317 template<typename... Types>
318 cv::GCompileArgs getCompileArgs(const std::vector<char> &sArgs) {
319  cv::GCompileArgs args;
320 
321  std::unique_ptr<cv::gapi::s11n::IIStream> pIs = cv::gapi::s11n::detail::getInStream(sArgs);
322  cv::gapi::s11n::IIStream& is = *pIs;
323 
324  uint32_t sz = 0;
325  is >> sz;
326  for (uint32_t i = 0; i < sz; ++i) {
327  std::string tag;
328  is >> tag;
329 
330  std::vector<char> sArg;
331  is >> sArg;
332 
333  cv::util::optional<GCompileArg> dArg =
334  cv::gapi::detail::tryDeserializeCompArg<Types...>(tag, sArg);
335 
336  if (dArg.has_value())
337  {
338  args.push_back(dArg.value());
339  }
340  }
341 
342  return args;
343 }

345 template<typename RMatAdapterType>
346 cv::GRunArgs getRunArgsWithRMats(const std::vector<char> &p) {
347  std::unique_ptr<cv::gapi::s11n::IIStream> pIs = cv::gapi::s11n::detail::getInStream(p);
348  cv::gapi::s11n::IIStream& is = *pIs;
349  cv::GRunArgs args;
350 
351  uint32_t sz = 0;
352  is >> sz;
353  for (uint32_t i = 0; i < sz; ++i) {
354  uint32_t idx = 0;
355  is >> idx;
356  args.push_back(cv::gapi::detail::deserialize_runarg<RMatAdapterType>::exec(is, idx));
357  }
358 
359  return args;
