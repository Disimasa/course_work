607 void Mat::forEach_impl(const Functor& operation) {
608  if (false) {
609  operation(*reinterpret_cast<_Tp*>(0), reinterpret_cast<int*>(0));
610  // If your compiler fails in this line.
611  // Please check that your functor signature is
612  // (_Tp&, const int*) <- multi-dimensional

614  }

620  class PixelOperationWrapper :public ParallelLoopBody
621  {
622  public:
623  PixelOperationWrapper(Mat_<_Tp>* const frame, const Functor& _operation)
624  : mat(frame), op(_operation) {}
625  virtual ~PixelOperationWrapper(){}
626  // ! Overloaded virtual operator
627  // convert range call to row call.
628  virtual void operator()(const Range &range) const CV_OVERRIDE
629  {
630  const int DIMS = mat->dims;
631  const int COLS = mat->size[DIMS - 1];
632  if (DIMS <= 2) {
633  for (int row = range.start; row < range.end; ++row) {
634  this->rowCall2(row, COLS);
635  }
636  } else {
637  std::vector<int> idx(DIMS);
638  idx[DIMS - 2] = range.start - 1;
639 
640  for (int line_num = range.start; line_num < range.end; ++line_num) {
641  idx[DIMS - 2]++;
642  for (int i = DIMS - 2; i >= 0; --i) {
643  if (idx[i] >= mat->size[i]) {
644  idx[i - 1] += idx[i] / mat->size[i];
645  idx[i] %= mat->size[i];
646  continue; // carry-over;
647  }
648  else {
649  break;
650  }
651  }
652  this->rowCall(&idx[0], COLS, DIMS);
653  }
654  }
655  }
656  private:
657  Mat_<_Tp>* const mat;
658  const Functor op;
659  // ! Call operator for each elements in this row.
660  inline void rowCall(int* const idx, const int COLS, const int DIMS) const {
661  int &col = idx[DIMS - 1];
662  col = 0;
663  _Tp* pixel = &(mat->template at<_Tp>(idx));
664 
665  while (col < COLS) {
666  op(*pixel, const_cast<const int*>(idx));
667  pixel++; col++;
668  }
669  col = 0;
670  }
671  // ! Call operator for each elements in this row. 2d mat special version.
672  inline void rowCall2(const int row, const int COLS) const {
673  union Index{
674  int body[2];
675  operator const int*() const {
676  return reinterpret_cast<const int*>(this);
677  }
678  int& operator[](const int i) {
679  return body[i];
680  }
681  } idx = {{row, 0}};
682  // Special union is needed to avoid
683  // "error: array subscript is above array bounds [-Werror=array-bounds]"
684  // when call the functor `op` such that access idx[3].
685 
686  _Tp* pixel = &(mat->template at<_Tp>(idx));
687  const _Tp* const pixel_end = pixel + COLS;
688  while(pixel < pixel_end) {
689  op(*pixel++, static_cast<const int*>(idx));
690  idx[1]++;
691  }
692  }
693  PixelOperationWrapper& operator=(const PixelOperationWrapper &) {
694  CV_Assert(false);
695  // We can not remove this implementation because Visual Studio warning C4822.
696  return *this;
697  }
698  };
699 
700  parallel_for_(cv::Range(0, LINES), PixelOperationWrapper(reinterpret_cast<Mat_<_Tp>*>(this), operation));
701 }

1042 template<class OBJECT>
1043 class CV_EXPORTS Node
1044 {
1045 public:
1046  Node()
1047  {
1048  m_pParent = 0;
1049  }
1050  Node(OBJECT& payload) : m_payload(payload)
1051  {
1052  m_pParent = 0;
1053  }
1054  ~Node()
1055  {
1056  removeChilds();
1057  if (m_pParent)
1058  {
1059  int idx = m_pParent->findChild(this);
1060  if (idx >= 0)
1061  m_pParent->m_childs.erase(m_pParent->m_childs.begin() + idx);
1062  }
1063  }
1064 
1065  Node<OBJECT>* findChild(OBJECT& payload) const
1066  {
1067  for(size_t i = 0; i < this->m_childs.size(); i++)
1068  {
1069  if(this->m_childs[i]->m_payload == payload)
1070  return this->m_childs[i];
1071  }
1072  return NULL;
1073  }
1074 
1075  int findChild(Node<OBJECT> *pNode) const
1076  {
1077  for (size_t i = 0; i < this->m_childs.size(); i++)
1078  {
1079  if(this->m_childs[i] == pNode)
1080  return (int)i;
1081  }
1082  return -1;
1083  }
1084 
1085  void addChild(Node<OBJECT> *pNode)
1086  {
1087  if(!pNode)
1088  return;
1089 
1090  CV_Assert(pNode->m_pParent == 0);
1091  pNode->m_pParent = this;
1092  this->m_childs.push_back(pNode);
1093  }
1094 
1095  void removeChilds()
1096  {
1097  for(size_t i = 0; i < m_childs.size(); i++)
1098  {
1099  m_childs[i]->m_pParent = 0; // avoid excessive parent vector trimming
1100  delete m_childs[i];
1101  }
1102  m_childs.clear();
1103  }
1104 
1105  int getDepth()
1106  {
1107  int count = 0;
1108  Node *pParent = m_pParent;
1109  while(pParent) count++, pParent = pParent->m_pParent;
1110  return count;
1111  }
1112 
1113 public:
1114  OBJECT m_payload;
1115  Node<OBJECT>* m_pParent;
1116  std::vector<Node<OBJECT>*> m_childs;
1117 };
