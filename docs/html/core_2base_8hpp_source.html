403 template<typename _Tp, typename _AccTp> static inline
404 _AccTp normL2Sqr(const _Tp* a, int n)
405 {
406  _AccTp s = 0;
407  int i=0;
408 #if CV_ENABLE_UNROLLED
409  for( ; i <= n - 4; i += 4 )
410  {
411  _AccTp v0 = a[i], v1 = a[i+1], v2 = a[i+2], v3 = a[i+3];
412  s += v0*v0 + v1*v1 + v2*v2 + v3*v3;
413  }
414 #endif
415  for( ; i < n; i++ )
416  {
417  _AccTp v = a[i];
418  s += v*v;
419  }
420  return s;
421 }

424 _AccTp normL1(const _Tp* a, int n)
425 {
426  _AccTp s = 0;
427  int i = 0;
428 #if CV_ENABLE_UNROLLED
429  for(; i <= n - 4; i += 4 )
430  {
431  s += (_AccTp)cv_abs(a[i]) + (_AccTp)cv_abs(a[i+1]) +
432  (_AccTp)cv_abs(a[i+2]) + (_AccTp)cv_abs(a[i+3]);
433  }
434 #endif
435  for( ; i < n; i++ )
436  s += cv_abs(a[i]);
