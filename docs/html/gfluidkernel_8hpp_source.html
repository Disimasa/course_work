353 template<typename, typename, typename, bool UseScratch>
354 struct FluidCallHelper;
355 
356 template<typename Impl, typename... Ins, typename... Outs, bool UseScratch>
357 struct FluidCallHelper<Impl, std::tuple<Ins...>, std::tuple<Outs...>, UseScratch>
358 {
359  static_assert(all_satisfy<is_gmat_type, Outs...>::value, "return type must be GMat");
360  static_assert(contains<GMat, Ins...>::value, "input must contain at least one GMat");
361 
362  // Execution dispatcher ////////////////////////////////////////////////////
363  template<int... IIs, int... OIs>
364  static void call_impl(const cv::GArgs &in_args,
365  const std::vector<gapi::fluid::Buffer*> &out_bufs,
366  detail::Seq<IIs...>,
367  detail::Seq<OIs...>)
368  {
369  Impl::run(fluid_get_in<Ins>::get(in_args, IIs)..., *out_bufs[OIs]...);
370  }
371 
372  static void call(const cv::GArgs &in_args,
373  const std::vector<gapi::fluid::Buffer*> &out_bufs)
374  {
375  constexpr int numOuts = (sizeof...(Outs)) + (UseScratch ? 1 : 0);
376  call_impl(in_args, out_bufs,
377  typename detail::MkSeq<sizeof...(Ins)>::type(),
378  typename detail::MkSeq<numOuts>::type());
379  }
380 
381  // Scratch buffer initialization dispatcher ////////////////////////////////
382  static void init_scratch(const GMetaArgs &metas,
383  const cv::GArgs &in_args,
384  gapi::fluid::Buffer &b)
385  {
386  scratch_helper<UseScratch, Impl, Ins...>::help_init(metas, in_args, b);
387  }
388 
389  // Scratch buffer reset dispatcher /////////////////////////////////////////
390  static void reset_scratch(gapi::fluid::Buffer &scratch_buf)
391  {
392  scratch_helper<UseScratch, Impl, Ins...>::help_reset(scratch_buf);
393  }
394 
395  static gapi::fluid::BorderOpt getBorder(const GMetaArgs &metas, const cv::GArgs &in_args)
396  {
397  constexpr bool hasWindow = has_Window<Impl, const int>::value;
398 
399  // User must provide "init" callback if Window != 1
400  // TODO: move to constexpr if when we enable C++17
401  return get_border_helper<callCustomGetBorder<hasWindow, Impl>::value, Impl, Ins...>::help(metas, in_args);
402  }
403 
404  static int getWindow(const GMetaArgs &metas, const cv::GArgs &in_args)
405  {
406  constexpr bool callCustomGetWindow = !(has_Window<Impl, const int>::value);
407  return get_window_helper<callCustomGetWindow, Impl, Ins...>::help(metas, in_args);
408  }
409 };
