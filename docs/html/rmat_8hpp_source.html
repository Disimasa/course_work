25 // "Remote Mat", a general class which provides an abstraction layer over the data
26 // storage and placement (host, remote device etc) and allows to access this data.
27 //
28 // The device specific implementation is hidden in the RMat::Adapter class
29 //
30 // The basic flow is the following:
31 // * Backend which is aware of the remote device:
32 // - Implements own AdapterT class which is derived from RMat::Adapter
33 // - Wraps device memory into RMat via make_rmat utility function:
34 // cv::RMat rmat = cv::make_rmat<AdapterT>(args);
35 //
36 // * End user:
37 // - Writes the code which works with RMats without any knowledge of the remote device:
38 // void func(const cv::RMat& in_rmat, cv::RMat& out_rmat) {
39 // // Fetch input data from the device, get mapped memory for output
40 // cv::RMat::View in_view = in_rmat.access(Access::R);
41 // cv::RMat::View out_view = out_rmat.access(Access::W);
42 // performCalculations(in_view, out_view);
43 // // data from out_view is transferred to the device when out_view is destroyed
44 // }
45 class GAPI_EXPORTS RMat
46 {
47 public:
48  // A lightweight wrapper on image data:
49  // - Doesn't own the memory;
50  // - Doesn't implement copy semantics (it's assumed that a view is created each time
51  // wrapped data is being accessed);
52  // - Has an optional callback which is called when the view is destroyed.
53  class GAPI_EXPORTS View
54  {
55  public:
56  using DestroyCallback = std::function<void()>;
57  using stepsT = std::vector<size_t>;
58 
59  View() = default;
60  View(const GMatDesc& desc, uchar* data, const stepsT& steps = {}, DestroyCallback&& cb = nullptr);
61  View(const GMatDesc& desc, uchar* data, size_t step, DestroyCallback&& cb = nullptr);
62 
63  View(const View&) = delete;
64  View& operator=(const View&) = delete;
65  View(View&&) = default;
66  View& operator=(View&& v);
67  ~View() { if (m_cb) m_cb(); }
68 
69  cv::Size size() const { return m_desc.size; }
70  const std::vector<int>& dims() const { return m_desc.dims; }
71  int cols() const { return m_desc.size.width; }
72  int rows() const { return m_desc.size.height; }
73  int type() const;
74  int depth() const { return m_desc.depth; }
75  int chan() const { return m_desc.chan; }
76  size_t elemSize() const { return CV_ELEM_SIZE(type()); }
77 
78  template<typename T = uchar> T* ptr(int y = 0) {
79  return reinterpret_cast<T*>(m_data + step()*y);
80  }
81  template<typename T = uchar> const T* ptr(int y = 0) const {
82  return reinterpret_cast<T*>(m_data + step()*y);
83  }
84  template<typename T = uchar> T* ptr(int y, int x) {
85  return reinterpret_cast<T*>(m_data + step()*y + step(1)*x);
86  }
87  template<typename T = uchar> const T* ptr(int y, int x) const {
88  return reinterpret_cast<const T*>(m_data + step()*y + step(1)*x);
89  }
90  size_t step(size_t i = 0) const { GAPI_DbgAssert(i<m_steps.size()); return m_steps[i]; }
91  const stepsT& steps() const { return m_steps; }
92 
93  private:
94  GMatDesc m_desc;
95  uchar* m_data = nullptr;
96  stepsT m_steps = {0u};
97  DestroyCallback m_cb = nullptr;
98  };

100  enum class Access { R, W };
101  class Adapter
102  {
103  public:
104  virtual ~Adapter() = default;
105  virtual GMatDesc desc() const = 0;
106  // Implementation is responsible for setting the appropriate callback to
107  // the view when accessed for writing, to ensure that the data from the view
108  // is transferred to the device when the view is destroyed
109  virtual View access(Access) = 0;
110  virtual void serialize(cv::gapi::s11n::IOStream&) {
111  GAPI_Assert(false && "Generic serialize method should never be called for RMat adapter");
112  }
113  virtual void deserialize(cv::gapi::s11n::IIStream&) {
114  GAPI_Assert(false && "Generic deserialize method should never be called for RMat adapter");
115  }
116  };
117  using AdapterP = std::shared_ptr<Adapter>;
118 
119  RMat() = default;
120  RMat(AdapterP&& a) : m_adapter(std::move(a)) {}
121  GMatDesc desc() const { return m_adapter->desc(); }
122 
123  // Note: When accessed for write there is no guarantee that returned view
124  // will contain actual snapshot of the mapped device memory
125  // (no guarantee that fetch from a device is performed). The only
126  // guaranty is that when the view is destroyed, its data will be
127  // transferred to the device
128  View access(Access a) const { return m_adapter->access(a); }
129 
130  // Cast underlying RMat adapter to the particular adapter type,
131  // return nullptr if underlying type is different
132  template<typename T> T* get() const
133  {
134  static_assert(std::is_base_of<Adapter, T>::value, "T is not derived from Adapter!");
135  GAPI_Assert(m_adapter != nullptr);
136  return dynamic_cast<T*>(m_adapter.get());
137  }
138 
139  void serialize(cv::gapi::s11n::IOStream& os) const {
140  m_adapter->serialize(os);
141  }
142 
143 private:
144  AdapterP m_adapter = nullptr;
145 };
