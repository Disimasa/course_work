213 template<typename T>
214 inline cv::util::optional<T> getCompileArg(const cv::GCompileArgs &args)
215 {
216  for (auto &compile_arg : args)
217  {
218  if (compile_arg.tag == cv::detail::CompileArgTag<T>::tag())
219  {
220  return cv::util::optional<T>(compile_arg.get<T>());
221  }
222  }
223  return cv::util::optional<T>();
224 }
