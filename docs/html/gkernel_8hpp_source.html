99  // Helper classes which brings outputMeta() marshalling to kernel
100  // implementations
101  //
102  // 1. MetaType establishes G#Type -> G#Meta mapping between G-API dynamic
103  // types and its metadata descriptor types.
104  // This mapping is used to transform types to call outMeta() callback.
105  template<typename T> struct MetaType;
106  template<> struct MetaType<cv::GMat> { using type = GMatDesc; };
107  template<> struct MetaType<cv::GMatP> { using type = GMatDesc; };
108  template<> struct MetaType<cv::GFrame> { using type = GFrameDesc; };
109  template<> struct MetaType<cv::GScalar> { using type = GScalarDesc; };
110  template<typename U> struct MetaType<cv::GArray<U> > { using type = GArrayDesc; };
111  template<typename U> struct MetaType<cv::GOpaque<U> > { using type = GOpaqueDesc; };
112  template<typename T> struct MetaType { using type = T; }; // opaque args passed as-is
113  // FIXME: Move it to type traits?
114 
115  // 2. Hacky test based on MetaType to check if we operate on G-* type or not
116  template<typename T> using is_nongapi_type = std::is_same<T, typename MetaType<T>::type>;
117 
118  // 3. Two ways to transform input arguments to its meta - for G-* and non-G* types:
119  template<typename T>
120  typename std::enable_if<!is_nongapi_type<T>::value, typename MetaType<T>::type>
121  ::type get_in_meta(const GMetaArgs &in_meta, const GArgs &, int idx)
122  {
123  return util::get<typename MetaType<T>::type>(in_meta.at(idx));
124  }
125 
126  template<typename T>
127  typename std::enable_if<is_nongapi_type<T>::value, T>
128  ::type get_in_meta(const GMetaArgs &, const GArgs &in_args, int idx)
129  {
130  return in_args.at(idx).template get<T>();
131  }
132 
133  // 4. The MetaHelper itself: an entity which generates outMeta() call
134  // based on kernel signature, with arguments properly substituted.
135  // 4.1 - case for multiple return values
136  // FIXME: probably can be simplified with std::apply or analogue.
137  template<typename, typename, typename>
138  struct MetaHelper;
139 
140  template<typename K, typename... Ins, typename... Outs>
141  struct MetaHelper<K, std::tuple<Ins...>, std::tuple<Outs...> >
142  {
143  template<int... IIs, int... OIs>
144  static GMetaArgs getOutMeta_impl(const GMetaArgs &in_meta,
145  const GArgs &in_args,
146  detail::Seq<IIs...>,
147  detail::Seq<OIs...>)
148  {
149  // FIXME: decay?
150  using R = std::tuple<typename MetaType<Outs>::type...>;
151  const R r = K::outMeta( get_in_meta<Ins>(in_meta, in_args, IIs)... );
152  return GMetaArgs{ GMetaArg(std::get<OIs>(r))... };
153  }
154  // FIXME: help users identify how outMeta must look like (via default impl w/static_assert?)
155 
156  static GMetaArgs getOutMeta(const GMetaArgs &in_meta,
157  const GArgs &in_args)
158  {
159  return getOutMeta_impl(in_meta,
160  in_args,
161  typename detail::MkSeq<sizeof...(Ins)>::type(),
162  typename detail::MkSeq<sizeof...(Outs)>::type());
163  }
164  };
165 
166  // 4.1 - case for a single return value
167  // FIXME: How to avoid duplication here?
168  template<typename K, typename... Ins, typename Out>
169  struct MetaHelper<K, std::tuple<Ins...>, Out >
170  {
171  template<int... IIs>
172  static GMetaArgs getOutMeta_impl(const GMetaArgs &in_meta,
173  const GArgs &in_args,
174  detail::Seq<IIs...>)
175  {
176  // FIXME: decay?
177  using R = typename MetaType<Out>::type;
178  const R r = K::outMeta( get_in_meta<Ins>(in_meta, in_args, IIs)... );
179  return GMetaArgs{ GMetaArg(r) };
180  }
181  // FIXME: help users identify how outMeta must look like (via default impl w/static_assert?)
182 
183  static GMetaArgs getOutMeta(const GMetaArgs &in_meta,
184  const GArgs &in_args)
185  {
186  return getOutMeta_impl(in_meta,
187  in_args,
188  typename detail::MkSeq<sizeof...(Ins)>::type());
189  }
190  };
