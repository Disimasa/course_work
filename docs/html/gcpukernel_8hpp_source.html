298 template<typename Impl, typename... Ins>
299 struct OCVSetupHelper<Impl, std::tuple<Ins...>>
300 {
301  // Using 'auto' return type and 'decltype' specifier in both 'setup_impl' versions
302  // to check existence of required 'Impl::setup' functions.
303  // While 'decltype' specifier accepts expression we pass expression with 'comma-operator'
304  // where first operand of comma-operator is call attempt to desired 'Impl::setup' and
305  // the second operand is 'void()' expression.
306  //
307  // SFINAE for 'Impl::setup' which accepts compile arguments.
308  template<int... IIs>
309  static auto setup_impl(const GMetaArgs &metaArgs, const GArgs &args,
310  GArg &state, const GCompileArgs &compileArgs,
311  detail::Seq<IIs...>) ->
312  decltype(Impl::setup(detail::get_in_meta<Ins>(metaArgs, args, IIs)...,
313  std::declval<typename std::add_lvalue_reference<
314  std::shared_ptr<typename Impl::State>
315  >::type
316  >(),
317  compileArgs)
318  , void())
319  {
320  // TODO: unique_ptr <-> shared_ptr conversion ?
321  // To check: Conversion is possible only if the state which should be passed to
322  // 'setup' user callback isn't required to have previous value
323  std::shared_ptr<typename Impl::State> stPtr;
324  Impl::setup(detail::get_in_meta<Ins>(metaArgs, args, IIs)..., stPtr, compileArgs);
325  state = GArg(stPtr);
326  }
327 
328  // SFINAE for 'Impl::setup' which doesn't accept compile arguments.
329  template<int... IIs>
330  static auto setup_impl(const GMetaArgs &metaArgs, const GArgs &args,
331  GArg &state, const GCompileArgs &/* compileArgs */,
332  detail::Seq<IIs...>) ->
333  decltype(Impl::setup(detail::get_in_meta<Ins>(metaArgs, args, IIs)...,
334  std::declval<typename std::add_lvalue_reference<
335  std::shared_ptr<typename Impl::State>
336  >::type
337  >()
338  )
339  , void())
340  {
341  // The same comment as in 'setup' above.
342  std::shared_ptr<typename Impl::State> stPtr;
343  Impl::setup(detail::get_in_meta<Ins>(metaArgs, args, IIs)..., stPtr);
344  state = GArg(stPtr);
345  }
346 
347  static void setup(const GMetaArgs &metaArgs, const GArgs &args,
348  GArg& state, const GCompileArgs &compileArgs)
349  {
350  setup_impl(metaArgs, args, state, compileArgs,
351  typename detail::MkSeq<sizeof...(Ins)>::type());
352  }
353 };

355 // OCVCallHelper is a helper class to call stateless OCV kernels and OCV kernel functors.
356 template<typename, typename, typename>
357 struct OCVCallHelper;
358 
359 // FIXME: probably can be simplified with std::apply or analogue.
360 template<typename Impl, typename... Ins, typename... Outs>
361 struct OCVCallHelper<Impl, std::tuple<Ins...>, std::tuple<Outs...>>
362 {
363  template<typename... Inputs>
364  struct call_and_postprocess
365  {
366  template<typename... Outputs>
367  static void call(Inputs&&... ins, Outputs&&... outs)
368  {
369  //not using a std::forward on outs is deliberate in order to
370  //cause compilation error, by trying to bind rvalue references to lvalue references
371  Impl::run(std::forward<Inputs>(ins)..., outs...);
372  postprocess(outs...);
373  }
374 
375  template<typename... Outputs>
376  static void call(Impl& impl, Inputs&&... ins, Outputs&&... outs)
377  {
378  impl(std::forward<Inputs>(ins)..., outs...);
379  }
380  };
381 
382  template<int... IIs, int... OIs>
383  static void call_impl(GCPUContext &ctx, detail::Seq<IIs...>, detail::Seq<OIs...>)
384  {
385  //Make sure that OpenCV kernels do not reallocate memory for output parameters
386  //by comparing it's state (data ptr) before and after the call.
387  //This is done by converting each output Mat into tracked_cv_mat object, and binding
388  //them to parameters of ad-hoc function
389  call_and_postprocess<decltype(get_in<Ins>::get(ctx, IIs))...>
390  ::call(get_in<Ins>::get(ctx, IIs)..., get_out<Outs>::get(ctx, OIs)...);
391  }
392 
393  template<int... IIs, int... OIs>
394  static void call_impl(cv::GCPUContext &ctx, Impl& impl,
395  detail::Seq<IIs...>, detail::Seq<OIs...>)
396  {
397  call_and_postprocess<decltype(get_in<Ins>::get(ctx, IIs))...>
398  ::call(impl, get_in<Ins>::get(ctx, IIs)..., get_out<Outs>::get(ctx, OIs)...);
399  }
400 
401  static void call(GCPUContext &ctx)
402  {
403  call_impl(ctx,
404  typename detail::MkSeq<sizeof...(Ins)>::type(),
405  typename detail::MkSeq<sizeof...(Outs)>::type());
406  }
407 
408  // NB: Same as call but calling the object
409  // This necessary for kernel implementations that have a state
410  // and are represented as an object
411  static void callFunctor(cv::GCPUContext &ctx, Impl& impl)
412  {
413  call_impl(ctx, impl,
414  typename detail::MkSeq<sizeof...(Ins)>::type(),
415  typename detail::MkSeq<sizeof...(Outs)>::type());
416  }
417 };

439  static void call_impl(GCPUContext &ctx, detail::Seq<IIs...>, detail::Seq<OIs...>)

443  ::call(st, get_in<Ins>::get(ctx, IIs)..., get_out<Outs>::get(ctx, OIs)...);
