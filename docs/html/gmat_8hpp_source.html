145  // Meta combinator: return a new GMatDesc with specified data depth.
146  // (all other fields are taken unchanged from this GMatDesc)
147  GMatDesc withDepth(int ddepth) const

155  // Meta combinator: return a new GMatDesc with specified data depth
156  // and number of channels.
157  // (all other fields are taken unchanged from this GMatDesc)
158  GMatDesc withType(int ddepth, int dchan) const
159  {
160  GAPI_Assert(CV_MAT_CN(ddepth) == 1 || ddepth == -1);
161  GMatDesc desc = withDepth(ddepth);

164  }

166  // Meta combinator: return a new GMatDesc with planar flag set
167  // (no size changes are performed, only channel interpretation is changed
168  // (interleaved -> planar)
169  GMatDesc asPlanar() const
170  {
171  GAPI_Assert(planar == false);
172  GMatDesc desc(*this);
173  desc.planar = true;
174  return desc;
175  }
176 
177  // Meta combinator: return a new GMatDesc
178  // reinterpreting 1-channel input as planar image
179  // (size height is divided by plane number)
180  GMatDesc asPlanar(int planes) const
181  {
182  GAPI_Assert(planar == false);
183  GAPI_Assert(chan == 1);
184  GAPI_Assert(planes > 1);
185  GAPI_Assert(size.height % planes == 0);
186  GMatDesc desc(*this);
187  desc.size.height /= planes;
188  desc.chan = planes;
189  return desc.asPlanar();
190  }
191 
192  // Meta combinator: return a new GMatDesc with planar flag set to false
193  // (no size changes are performed, only channel interpretation is changed
194  // (planar -> interleaved)
195  GMatDesc asInterleaved() const
196  {
197  GAPI_Assert(planar == true);
198  GMatDesc desc(*this);
199  desc.planar = false;
200  return desc;
201  }
