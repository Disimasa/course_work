103  // Tests if Trait for type T requires extra marshalling ("custom wrap") or not.

108  template<typename C> static std::true_type test(check<C, decltype(&GTypeTraits<C>::wrap_value)> *);

135  // Marshalling helper for G-types and its Host types. Helps G-API
136  // to store G types in internal generic containers for further
137  // processing. Implements the following callbacks:
138  //
139  // * wrap() - converts user-facing G-type into an internal one
140  // for internal storage.
141  // Used when G-API operation is instantiated (G<Kernel>::on(),
142  // etc) during expressing a pipeline. Mostly returns input
143  // value "as is" except the case when G-type is a template. For
144  // template G-classes, calls custom wrap() from Traits.
145  // The value returned by wrap() is then wrapped into GArg() and
146  // stored in G-API metadata.
147  //
148  // Example:
149  // - cv::GMat arguments are passed as-is.
150  // - integers, pointers, STL containers, user types are passed as-is.
151  // - cv::GArray<T> is converted to cv::GArrayU.
152  //
153  // * wrap_in() / wrap_out() - convert Host type associated with
154  // G-type to internal representation type.
155  //
156  // - For "simple" (non-template) G-types, returns value as-is.
157  // Example: cv::GMat has host type cv::Mat, when user passes a
158  // cv::Mat, system stores it internally as cv::Mat.
159  //
160  // - For "complex" (template) G-types, utilizes custom
161  // wrap_in()/wrap_out() as described in Traits.
162  // Example: cv::GArray<T> has host type std::vector<T>, when
163  // user passes a std::vector<T>, system stores it
164  // internally as VectorRef (with <T> stripped away).
165  template<typename T, class Custom = void> struct WrapValue
166  {
167  static auto wrap(const T& t) ->
168  typename std::remove_reference<T>::type
169  {
170  return static_cast<typename std::remove_reference<T>::type>(t);
171  }
172 
173  template<typename U> static U wrap_in (const U &u) { return u; }
174  template<typename U> static U* wrap_out(U &u) { return &u; }
175  };
176  template<typename T> struct WrapValue<T, typename std::enable_if<has_custom_wrap<T>::value>::type>
177  {
178  static auto wrap(const T& t) -> decltype(GTypeTraits<T>::wrap_value(t))
179  {
180  return GTypeTraits<T>::wrap_value(t);
181  }
182  template<typename U> static auto wrap_in (const U &u) -> typename GTypeTraits<T>::strip_type
183  {
184  return GTypeTraits<T>::wrap_in(u);
185  }
186  template<typename U> static auto wrap_out(U &u) -> typename GTypeTraits<T>::strip_type
187  {
188  return GTypeTraits<T>::wrap_out(u);
189  }
190  };
